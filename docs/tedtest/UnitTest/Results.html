259    def pretty(
260        self, filter: list[TestFilter] = [TestFilter.NONE], indent: int = 0
261    ) -> list:
262        """Used to convert results into a list of strings. Allows for additional indentation to be added.
263
264        Args:
265            indent (int, optional): Amount of indent to add in spaces. Defaults to 0.
266
267        Returns:
268            list: The formatted results as a list of string with indents
269        """
270        out = []
271        if TestFilter.NONE not in filter:
272            if TestFilter.OVERALL in filter or self.result.filter in filter:
273                out.append(
274                    "".ljust(indent, " ")
275                    + f"\[{self.color}{self.symbol}[@F]] {TED.encode(self.name)}"
276                )
277                if isinstance(self.info, list):
278                    for trace in self.info:
279                        out.append("".ljust(indent + 4, " ") + trace)
280                else:
281                    if self.info != "":
282                        out.append("".ljust(indent + 4, " ") + self.info)
283
284        return out
285
286    def str(self, indent: int = 0) -> list:
287        """Results formatted into lines without colors
288
289        Args:
290            indent (int, optional): The amount to indent the lines. Defaults to 0.
291
292        Returns:
293            list: The results as lines
294        """
295        out = []
296        out.append(" " * indent + f"[{self.symbol}] {self.name}")
297        if isinstance(self.info, list):
298            for trace in self.info:

300        else:
301            if self.info != "":

304        return out

354        """Saves the data to a given file type in a given location

356        Args:
357            location (str): Where to save the file
358            type (str): Type of file to save, CSV, JSON, TXT
359
360        Returns:
361            bool: True if file was saved
362        """
